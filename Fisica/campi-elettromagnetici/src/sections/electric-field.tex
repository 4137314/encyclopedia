\section{Electric Field}

\subsection{Charge}
Line Charge Density

\begin{aligned}
	\rho_l &= \lim_{\Delta l \to 0} \frac{\Delta Q}{\Delta l} \quad &\mathrm{\left[ \frac{C}{m}\right]}
\end{aligned}

Surface Charge Density

\begin{aligned}
\rho _s =\lim_{\Delta s \to 0} \frac{\Delta Q}{\Delta s} \quad & \mathrm{\left[\frac{C}{m^2}\right]} 
\end{aligned}

Volume Charge Density

\begin{aligned}
	\rho _v = \lim_{\Delta v \to 0} \frac{\Delta Q}{\Delta v} \quad & \mathrm{\left[\frac{C}{^3}\right]}
\end{aligned}

\subsection{Coulomb's Law}
\begin{aligned}
	\vec F= k \frac{Q_1 Q_2}{r^2} \hat r \quad & \mathrm{[N]}
\end{aligned}

\begin{aligned}
	k= \frac{1}{4 \pi \epsilon} \quad & \mathrb{\left[\frac{N m^2}{C^2}\right]}
\end{aligned}
\subsection{Electric Field}
\begin{aligned}
	\vec E = \lim_{Q \to 0} \frac{\vec F}{Q} \quad & \mathrb{\left[\frac{N}{C}\right]}
\end{aligned}
\subsection{Electric Flux}
Electric Flux Density

\begin{aligned}
	\vec D = \epsilo \vec E \quad & \mathrb{\left[\frac{C}{m^2}\right]}	
\end{aligned}

Electric Flux

\begin{aligned}
	\Phi = \int _S \vec D \cdot d \vec s \quad \mathrb{[C]}
\end{aligned}
